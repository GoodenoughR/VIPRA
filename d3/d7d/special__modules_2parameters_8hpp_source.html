93 std::string const& paramName, Random::Engine& engine) const
94 -> std::remove_cvref_t<data_t>
95 {
96 using param_t = std::remove_cvref_t<data_t>;
97
98 std::string moduleStr = to_string(module);
99
100 // Check that the parameter was registered
101 if ( ! contains(module, moduleName, paramName) ) {
102 throw std::runtime_error("Parameter: " + paramName + " For " + to_string(module) +
103 " Module: " + moduleName + " Not Registered");
104 }
105
106 // load the value of the parameter
107 auto value = std::optional<param_t>{};
108 if constexpr ( Util::is_specialization<param_t, std::vector>::value ) {
109 // the parameter is supposed to be an array, load it as such
110 value = get_array_param<param_t>(module, moduleName, paramName);
111 }
112 else {
113 // the parameter is meant to be a single value, follow parameter randomization rules
114 value = _input.template get_param<param_t>(moduleStr, moduleName, paramName, engine);
115 }
116
117 if ( ! value.has_value() ) {
118 // parameter value wasn't found, throw
119 throw std::runtime_error("Required Parameter: " + paramName + " For " + to_string(module) +
120 " Module: " + moduleName + " Not Provided In Input");
121 }
122

126 // });

128 return param_t{std::move(value.value())};

141 void set_end_goals(auto const& pedset, auto const& map)
142 {
143 assert(pedset.num_pedestrians() > 0);
144
145 const VIPRA::size pedCnt = pedset.num_pedestrians();
146
147 // find the end goals, provided as a module parameter
148 auto const& objects = map.get_objects(_endGoalType);
149 if ( objects.empty() ) {
150 throw std::runtime_error("No objects of type " + _endGoalType + " found in map");
151 }
152
153 // set each pedestrians end goal as the nearest of that type
154 for ( VIPRA::idx pedIdx = 0; pedIdx < pedCnt; ++pedIdx ) {

157 auto const nearestGoal = get_nearest_goal(pos, objects);

174 [[nodiscard]] static auto squash_path(std::vector<VIPRA::f3d> const& path) -> std::vector<VIPRA::f3d>
175 {
176 assert(path.empty() == false);
177
178 std::vector<VIPRA::f3d> squashedPath;
179
180 VIPRA::f3d dif;
181 for ( VIPRA::idx i = 1; i < path.size(); ++i ) {
182 auto currDif = path[i] - path[i - 1];
183 if ( currDif != dif ) {
184 squashedPath.push_back(path[i - 1]);
185 dif = currDif;
186 }
187 }
188
189 return squashedPath;
190 }

199 [[nodiscard]] static auto get_nearest_goal(VIPRA::f3d pos, std::vector<VIPRA::f3d> const& goals)
200 -> VIPRA::f3dVec::const_iterator
201 {
202 assert(goals.empty() == false);
203
204 return std::min_element(goals.begin(), goals.end(), [&](auto const& left, auto const& right) {
205 return pos.distance_to(left) < pos.distance_to(right);

215 void find_path(VIPRA::idx pedIdx, VIPRA::f3d startPos)
216 {
217 // Get the pedestrian start and end grid location
218 VIPRA::idx startIdx = _graph.get_closest_grid_idx(startPos);
219 VIPRA::idx endIdx = _graph.get_closest_grid_idx(_endGoals[pedIdx]);
220 if ( _graph.nodes().size() <= startIdx || _graph.nodes().size() <= endIdx ) {
221 throw std::runtime_error("Start or end index is out of bounds");
222 }
223
224 // use A* to find the path
225 auto const path = VIPRA::Algo::astar(
226 startIdx, endIdx, _graph,
227 [&](VIPRA::idx left, VIPRA::idx right) -> VIPRA::f_pnt {
228 return _graph.data(left).pos.distance_to(_graph.data(right).pos);
229 },
230 [&](VIPRA::idx nodeIdx) -> VIPRA::f3d { return _graph.data(nodeIdx).pos; });
231
232 if ( ! path ) {
233 // No path found
234 throw std::runtime_error("No path found for pedestrian");
235 }
236
237 // set their path, squash nodes that all go in the same direction into one node
238 _paths[pedIdx] = squash_path(*path);
239 _currentGoals[pedIdx] = _paths[pedIdx].back();
240 }

113

117 openset.push(&nodes[start]);

119 while (!openset.empty()) {
120 current = openset.top();

122 if (current->self == end) {

126 openset.pop();
127 closedset.insert(current);
128 for (VIPRA::idx neighborIdx : graph.neighbors(current->self)) {
129 if (!closedset.contains(&nodes[neighborIdx])) {

131 neighbor.self = neighborIdx;
132 neighbor.parent = current->self;
133 neighbor.distanceFromStart = current->distanceFromStart + distance_func(current->self, neighborIdx);
134 neighbor.distanceWithHeuristic = neighbor.distanceFromStart + distance_func(neighborIdx, end);
135
136 auto found = std::find(openset.begin(), openset.end(), &nodes[neighborIdx]);
137 if (found == openset.end()) {
138 nodes[neighborIdx] = neighbor;
139 openset.push(&nodes[neighborIdx]);
140 } else {
141 if (neighbor.distanceFromStart < (*found)->distanceFromStart) {
142 (*found)->distanceFromStart = neighbor.distanceFromStart;
143 (*found)->distanceWithHeuristic = neighbor.distanceWithHeuristic;
144 (*found)->parent = neighbor.parent;
145 }
146 }
147 }
148 }
149 }
150
151 if (current->self != end) {
152 return std::nullopt;
153 }
154

156 while (current->self != start) {
157 if constexpr (std::is_same_v<conversion_f_t, VOID>) {
158 path.push_back(current->self);
159 } else {
160 path.push_back(conversion_func(current->self));
161 }
162
163 current = &nodes[current->parent];
164 }
165
166 if constexpr (std::is_same_v<conversion_f_t, VOID>) {
167 path.push_back(start);
168 } else {
169 path.push_back(conversion_func(start));
170 }

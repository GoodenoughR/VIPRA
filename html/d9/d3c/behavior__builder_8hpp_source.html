240 template <typename comp_t>
241 [[nodiscard]] static auto find_event_component(BehaviorParser::EventContext* ctx) -> std::optional<comp_t> {
242 for (auto const& attr : ctx->event_attribute()) {
243 // Name
244 if constexpr (std::is_same_v<comp_t, std::string>)
245 if (attr->event_name()) return attr->event_name()->ID()->toString();
246
247 // Start Condition
248 if constexpr (std::is_same_v<comp_t, evStart>)
249 if (attr->event_start()) return attr->event_start();
250
251 // End Condition
252 if constexpr (std::is_same_v<comp_t, evEnd>)
253 if (attr->event_end()) return attr->event_end();
254 }
255
256 return std::nullopt;
257 }
258

267 [[nodiscard]] static auto find_action_component(BehaviorParser::ActionContext* ctx)
268 -> std::optional<comp_t> {
269 for (auto const& attr : ctx->action_attribute()) {
270 // Stimulus

273
274 // Response
275 if constexpr (std::is_same_v<comp_t, acResponse>)
276 if (attr->action_response()) return attr->action_response();
277
278 // Target
279 if constexpr (std::is_same_v<comp_t, acTarget>)
280 if (attr->action_target()) return attr->action_target();
281
282 // Duration

284 if (attr->action_duration()) return attr->action_duration();

355template <Concepts::PedsetModule pedset_t, Concepts::MapModule map_t, Concepts::GoalsModule goals_t>
356auto BehaviorBuilder<pedset_t, map_t, goals_t>::build(std::string behaviorName,
357 std::filesystem::path const& filepath, seed seedNum)
358 -> HumanBehavior {
359 if (!std::filesystem::exists(filepath)) {
360 // spdlog::error("Behavior \"{}\" Does NOT Exist at {}", behaviorName, filepath.c_str());
361 BuilderException::error();
362 }
363
364 // spdlog::debug("Loading Behavior: {} at {}", behaviorName, std::filesystem::canonical(filepath).c_str());
365 initial_behavior_setup(behaviorName, seedNum);
366
367 std::ifstream dslFile(filepath);
368
369 antlr4::ANTLRInputStream input(dslFile);
370 BehaviorLexer lexer(&input);
371 antlr4::CommonTokenStream tokens(&lexer);
372 BehaviorParser parser(&tokens);
373
374 parser.removeErrorListeners();
375 parser.addErrorListener(&_errorListener);
376
377 BehaviorParser::ProgramContext* tree = parser.program();
378
379 visitProgram(tree);
380
381 end_behavior_check();
382
383 return _currentBehavior;
384}

1057template <Concepts::PedsetModule pedset_t, Concepts::MapModule map_t, Concepts::GoalsModule goals_t>
1058auto BehaviorBuilder<pedset_t, map_t, goals_t>::visitEvent(BehaviorParser::EventContext* ctx)
1059 -> antlrcpp::Any {
1060 auto name = find_event_component<evName>(ctx);
1061 if (!name) error(R"(Behavior "{}": Missing Event Name)", _currentBehavior.get_name());
1062 std::string eventName = "!" + name.value();
1063
1064 if (get_event(eventName))
1065 error(R"(Behavior "{}": Attempt To Redefine Event: "{}")", _currentBehavior.get_name(), eventName);
1066
1067 auto startCond = find_event_component<evStart>(ctx);
1068 if (!startCond) error(R"(Behavior "{}": Missing Start Condition)", _currentBehavior.get_name());
1069
1070 auto endCond = find_event_component<evEnd>(ctx);
1071
1072 // NOTE(rolland): a bit odd but we need to ensure that the event is in the map, incase the conditions rely on it
1073 _eventsMap[eventName] = _currentBehavior.event_count();
1074
1075 Event event(eventName);
1076
1077 // spdlog::debug(R"(Behavior "{}": Event: "{}" Adding Start Condition)", _currentBehavior.get_name(),
1078 // eventName);
1079 event.set_start_condition(build_condition(startCond.value()->condition()));
1080
1081 if (endCond) {
1082 // spdlog::debug(R"(Behavior "{}": Event: "{}" Adding End Condition)", _currentBehavior.get_name(),
1083 // eventName);
1084 event.set_end_condition(build_condition(endCond.value()->condition()));
1085 }
1086
1087 _eventsMap[eventName] = _currentBehavior.add_event(event);
1088
1089 return BehaviorBaseVisitor::visitEvent(ctx);
1090}
1091template <Concepts::PedsetModule pedset_t, Concepts::MapModule map_t, Concepts::GoalsModule goals_t>
1092auto BehaviorBuilder<pedset_t, map_t, goals_t>::visitAction(BehaviorParser::ActionContext* ctx)
1093 -> antlrcpp::Any {
1094 auto response = find_action_component<acResponse>(ctx);
1095 if (!response) error(R"(Behavior "{}": Action has no Response)", _currentBehavior.get_name());
1096
1097 auto stimulus = find_action_component<acStimulus>(ctx);
1098 auto target = find_action_component<acTarget>(ctx);
1099 auto duration = find_action_component<acDuration>(ctx);
1100
1101 Action action;
1102
1103 auto const typeStr = ctx->ID()->toString();
1104 auto const type = get_check_type(typeStr);
1105
1106 // spdlog::debug("Behavior \"{}\": Adding Action For {}", _currentBehavior.get_name(), typeStr);
1107
1108 auto atoms = response.value()->sub_action()->action_atom();
1109 std::for_each(atoms.begin(), atoms.end(),
1110 [&](BehaviorParser::Action_atomContext* atom) { add_atom_to_action(action, atom); });
1111
1112 if (stimulus) action.add_condition(build_condition(stimulus.value()->condition()));
1113 if (target) add_target_to_action(action, target.value()->target());
1114 if (duration) action.add_duration(get_numeric(duration.value()->duration()->value_numeric(), _currSeed));
1115
1116 _currentBehavior.add_action(type, action);
1117 return BehaviorBaseVisitor::visitAction(ctx);
1118}

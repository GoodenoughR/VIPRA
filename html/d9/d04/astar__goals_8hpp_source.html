58 startIdx, endIdx, _graph,

100 void update(pedset_t const& pedset, map_t const& /*unused*/, VIPRA::delta_t deltaT) {
101 assert(pedset.num_pedestrians() > 0);
102
103 for (VIPRA::idx pedIdx = 0; pedIdx < pedset.num_pedestrians(); ++pedIdx) {
104 auto const pos = pedset.ped_coords(pedIdx);
105 _timeSinceLastGoal[pedIdx] += deltaT;

107 if (!_paths[pedIdx].empty()) {
108 _currentGoals[pedIdx] = _paths[pedIdx].back();
109 _paths[pedIdx].pop_back();
110 _timeSinceLastGoal[pedIdx] = 0.0F;
111 }
112 }

182 void construct_graph(Concepts::MapModule auto const& map) {
183 set_grid_counts(map);
184
185 assert(_xCount > 0 && _yCount > 0);
186
187 _graph.reserve(_xCount * _yCount);
188
189 // TODO(rolland): This starts the grid a (0, 0) and not the bottom left corner of the map, we may want to change this
190 VIPRA::f3d center;
191
192 for (VIPRA::idx currY = 0; currY < _yCount; ++currY) {
193 center.x = 0.0F;
194 center.y += _grid_size;
195
196 for (VIPRA::idx currX = 0; currX < _xCount; ++currX) {
197 center.x += _grid_size;
198 const VIPRA::idx currIdx = _graph.add_node(
199 GridPoint{center, !map.collision(VIPRA::Geometry::Circle{center, _closest_obstacle})});
200
201 if (!_graph.data(currIdx).traversable) {
202 continue;
203 }
204
205 set_adjacents(currIdx, currX, currY);
206 }
207 }

217 // NOLINTNEXTLINE(bugprone-easily-swappable-parameters)
218 void set_adjacents(VIPRA::idx currIdx, VIPRA::idx currX, VIPRA::idx currY) {
219 assert(currIdx < _graph.nodes().size());
220 assert(currX < _xCount && currY < _yCount);
221 assert(_graph.data(currIdx).traversable == true);
222
223 if (currX > 1) {
224 if (_graph.data(currIdx - 1).traversable) _graph.add_edge(currIdx, currIdx - 1);
225 }
226 if (currY > 1) {
227 if (_graph.data(currIdx - _xCount).traversable) _graph.add_edge(currIdx, currIdx - _xCount);
228 }
229 if (currX > 1 && currY > 1) {
230 if (_graph.data(currIdx - _xCount - 1).traversable) _graph.add_edge(currIdx, currIdx - _xCount - 1);
231 }
232 if (currX < _xCount - 1 && currY > 1) {
233 if (_graph.data(currIdx - _xCount + 1).traversable) _graph.add_edge(currIdx, currIdx - _xCount + 1);
234 }
235 }

243 void set_end_goals(Concepts::PedsetModule auto const& pedset, Concepts::MapModule auto const& map) {
244 assert(pedset.num_pedestrians() > 0);
245
246 const VIPRA::size pedCnt = pedset.num_pedestrians();
247
248 auto const& objects = map.get_objects(_end_goal_type);
249 if (objects.empty()) {
250 throw std::runtime_error("No objects of type " + _end_goal_type + " found in map");
251 }
252
253 for (VIPRA::idx pedIdx = 0; pedIdx < pedCnt; ++pedIdx) {
254 auto const pos = pedset.ped_coords(pedIdx);
255
256 auto const nearestGoal = get_nearest_goal(pos, objects);
257
258 if (nearestGoal != objects.end()) {
259 _endGoals[pedIdx] = *nearestGoal;
260 } else {
261 throw std::runtime_error("No goal found for pedestrian");
262 }
263 }

272 [[nodiscard]] static auto squash_path(std::vector<VIPRA::f3d> const& path) -> std::vector<VIPRA::f3d> {
273 assert(path.empty() == false);
274
275 std::vector<VIPRA::f3d> squashedPath;
276 squashedPath.reserve(path.size() / 2);
277
278 VIPRA::f3d dif;
279 for (VIPRA::idx i = 1; i < path.size(); ++i) {
280 auto currDif = path[i] - path[i - 1];
281 if (currDif != dif) {
282 squashedPath.push_back(path[i - 1]);
283 dif = currDif;
284 }
285 }
286
287 return squashedPath;
288 }

310 [[nodiscard]] static auto get_nearest_goal(VIPRA::f3d pos, std::vector<VIPRA::f3d> const& goals)
311 -> VIPRA::f3dVec::const_iterator {
312 assert(goals.empty() == false);
313
314 return std::min_element(goals.begin(), goals.end(), [&](auto const& left, auto const& right) {
315 return pos.distance_to(left) < pos.distance_to(right);

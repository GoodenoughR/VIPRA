87 auto run_sim(Concepts::ParamModule auto&& params) -> output_data_t {
88 auto [maxTimestep, timestepSize, randomseed, state] = initialize(params);
89 _output.new_run(_currSimIdx++);
90 _timestep = 0;

92 while (_timestep < maxTimestep && !_goals.is_sim_goal_met()) {
93 _model.timestep(_pedset, _map, _goals, _output, state, timestepSize, _timestep);
94 _behaviorModel.timestep(_pedset, _map, _goals, state, timestepSize);

97 output_positions();

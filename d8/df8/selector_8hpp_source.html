108template <typename select_t, typename context_t>
109auto Selector<select_t, context_t>::select_peds_from_group(select_t& selector, auto pack,
110 std::string const& /*unused*/) -> VIPRA::idxVec
111{
112 auto const& fullGroup = _pedGroups.get_group(selector.group);
113 auto usablegroup = filter_used_peds(fullGroup, _pedGroups.get_used(selector.group));
114 auto result = selector.select_peds(fullGroup, usablegroup, pack);
115
116 if ( ! result.starved ) {
117 return result.group;
118 }
119
120 if ( selector.required ) {
121 // spdlog::error("Behavior: {}, Required Selector Starved For Type: {} From Group: {}", behaviorName,
122 // selector.type.fullType, selector.group);
123 throw std::runtime_error("");
124 }
125
126 // spdlog::warn("Behavior: {}, Starved Selector For Type: {} From Group: {}", behaviorName,
127 // selector.type.fullType, selector.group);
128 return result.group;
129}

213template <typename select_t, typename context_t>
214auto Selector<select_t, context_t>::order_selectors() -> VIPRA::idxVec
215{
216 VIPRA::idxVec order;
217
218 for ( VIPRA::idx selIdx = 0; selIdx < _subSelectors.size(); ++selIdx ) {
219 if ( _subSelectors[selIdx].group == 0 ) {
220 order.push_back(selIdx);
221 }
222 }
223
224 _allTypes.for_each_type([&](typeUID type) {
225 for ( VIPRA::idx selIdx = 0; selIdx < _subSelectors.size(); ++selIdx ) {
226 if ( _subSelectors[selIdx].group == type ) {
227 order.push_back(selIdx);
228 }
229 }
230 });
231
232 for ( VIPRA::idx selIdx = 0; selIdx < _subSelectors.size(); ++selIdx ) {
233 bool notInGraph = std::find_if(order.begin(), order.end(),
234 [&](VIPRA::idx index) { return index == selIdx; }) == order.end();
235 if ( notInGraph ) {
236 order.push_back(selIdx);
237 }
238 }
239
240 check_for_duplicates(order);
241 return order;
242}

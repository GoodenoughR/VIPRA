159 static void create_output_directory(std::filesystem::path const& directory)
160 {
161 if ( std::filesystem::exists(directory) ) {
162 if ( std::filesystem::is_directory(directory) ) {
163 // directory exists and is actually a directory, all is good
164 return;
165 }
166
167 // exists but isn't a directory, error
168 throw std::runtime_error("Output directory already exists and is not a directory: " +
169 directory.string());
170 }
171
172 // create and check it was actually created
173 if ( ! std::filesystem::create_directory(directory) ) {
174 if ( ! std::filesystem::exists(directory) )
175 throw std::runtime_error("Could not create output directory: " + directory.string());

118template <typename... key_ts>
119[[nodiscard]] auto VIPRA::Input::JSON::load_polygons(key_ts&&... keys) const
120 -> std::optional<std::vector<Geometry::Polygon>> {
121 assert(_loaded);
122
123 auto value = get_value_at_key(std::forward<key_ts>(keys)...);
124 if (!value) return std::nullopt;
125
126 std::vector<Geometry::Polygon> polygons;
127 for (auto const& polygon : value.value().get()) {
128 if (!polygon.is_array()) return std::nullopt;
129
130 std::vector<VIPRA::f3d> points;
131 for (auto const& point : polygon) {
132 if (!is_f2d(point)) return std::nullopt;
133 points.emplace_back(point.at("x").template get<VIPRA::f_pnt>(),
134 point.at("y").template get<VIPRA::f_pnt>(), 0);
135 }
136
137 polygons.emplace_back(points);
138 }
139
140 return polygons;
141}

291inline void VIPRA::Input::JSON::load() {
292 if (_loaded) return;
293
294 if (!std::filesystem::exists(_filepath))
295 throw std::runtime_error("File does not exist at: " + _filepath.string());
296
297 if (!std::filesystem::is_regular_file(_filepath))
298 throw std::runtime_error("File is not a regular file at: " + _filepath.string());
299
300 std::ifstream file(_filepath, std::ios::in);
301 if (!file.is_open()) throw std::runtime_error("Could not open file at: " + _filepath.string());
302
303 try {
304 _json = nlohmann::json::parse(file);
305 } catch (nlohmann::json::parse_error const& e) {
306 throw std::runtime_error("Could not parse JSON file at: " + _filepath.string() + "\n" + e.what());
307 }
308
309 _loaded = true;
310 file.close();
311}

313template <typename... key_ts>
314[[nodiscard]] auto VIPRA::Input::JSON::get_value_at_key(key_ts&&... keys) const
315 -> std::optional<std::reference_wrapper<const nlohmann::json>> {
316 auto value = std::cref(_json);
317 bool found = true;
318
319 auto const findKey = [&](std::string const& key) {
320 if (!found) return;
321
322 if (!value.get().contains(key)) {
323 found = false;
324 return;
325 }
326
327 value = std::cref(value.get().at(key));
328 };
329
330 (findKey(keys), ...);
331
332 if (!found) return std::nullopt;
333
334 return value;
335}

344template <typename data_t>
345[[nodiscard]] auto VIPRA::Input::JSON::get_value(
346 std::reference_wrapper<const nlohmann::json> const& value) const -> std::optional<data_t> {
347 if constexpr (std::is_same_v<data_t, VIPRA::Parameter<std::string>>) {
348 return get_parameter(value);
349 }
350
351 else if constexpr (std::is_same_v<data_t, VIPRA::f3d>) {
352 return get_f3d(value);
353 }
354
355 else if constexpr (std::is_same_v<data_t, VIPRA::f3dVec>) {
356 return get_f3d_vec(value);
357 }
358
359 else if constexpr (Util::is_specialization<data_t, std::vector>::value) {
360 using value_t = typename Util::get_specialization_internal<data_t>::type;
361 return get_vector<value_t>(value);
362 }
363
364 else if constexpr (Util::is_specialization<data_t, Parameter>::value) {
365 using value_t = typename Util::get_specialization_internal<data_t>::type;
366 return numeric_parameter_helper<value_t>(value);
367 } else {
368 try {
369 return value.get().get<data_t>();
370 } catch (nlohmann::json::type_error const& e) {
371 return std::nullopt;
372 }
373 }
374
375 return std::nullopt;
376}

385[[nodiscard]] inline auto VIPRA::Input::JSON::is_f3d(nlohmann::json const& value) -> bool {

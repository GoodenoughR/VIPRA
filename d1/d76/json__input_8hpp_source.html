144[[nodiscard]] inline auto VIPRA::Input::JSON::get_f3d_vec(json_cref const& value)
145 -> std::optional<VIPRA::f3dVec>
146{
147 if ( ! value.get().is_array() ) return std::nullopt;
148
149 VIPRA::f3dVec f3dVec;
150 for ( auto const& f3d : value.get() ) {

158 f3dVec.emplace_back(f3d.at("x").get<VIPRA::f_pnt>(), f3d.at("y").get<VIPRA::f_pnt>(),
159 f3d.at("z").get<VIPRA::f_pnt>());

184inline void VIPRA::Input::JSON::load_impl()
185{
186 if ( ! std::filesystem::exists(_filepath) )
187 throw std::runtime_error("File does not exist at: " + _filepath.string());
188
189 if ( ! std::filesystem::is_regular_file(_filepath) )
190 throw std::runtime_error("File is not a regular file at: " + _filepath.string());
191
192 std::ifstream file(_filepath, std::ios::in);
193 if ( ! file.is_open() ) throw std::runtime_error("Could not open file at: " + _filepath.string());
194
195 try {
196 _json = nlohmann::json::parse(file);
197 }
198 catch ( nlohmann::json::parse_error const& e ) {
199 throw std::runtime_error("Could not parse JSON file at: " + _filepath.string() + "\n" + e.what());
200 }
201
202 file.close();
203}

212template <typename... key_ts>
213[[nodiscard]] auto VIPRA::Input::JSON::get_value_at_key(key_ts&&... keys) const -> std::optional<json_cref>
214{
215 auto value = std::cref(_json);
216 bool found = true;
217
218 auto const findKey = [&](std::string_view key) {
219 if ( ! found ) return;
220
221 if ( ! value.get().contains(key) ) {
222 found = false;
223 return;
224 }
225
226 value = std::cref(value.get().at(key));
227 };
228
229 // drill through each key to get the final value
230 (findKey(keys), ...);
231
232 if ( ! found ) return std::nullopt;
233
234 return value;
235}

249inline auto VIPRA::Input::JSON::get_polygons(json_cref const& value)
250 -> std::optional<std::vector<VIPRA::Geometry::Polygon>>
251{
252 std::vector<Geometry::Polygon> polygons;
253 for ( auto const& polygon : value.get() ) {
254 if ( ! polygon.is_array() ) return std::nullopt;
255
256 std::vector<VIPRA::f3d> points;
257 for ( auto const& point : polygon ) {
258 if ( ! is_f2d(point) ) return std::nullopt;
259 points.emplace_back(point.at("x").template get<VIPRA::f_pnt>(),
260 point.at("y").template get<VIPRA::f_pnt>(), 0);
261 }
262
263 polygons.emplace_back(points);
264 }
265
266 return polygons;
267}

276template <typename data_t>
277[[nodiscard]] auto VIPRA::Input::JSON::get_value(json_cref const& value) const -> std::optional<data_t>
278{
279 if constexpr ( std::is_same_v<data_t, VIPRA::f3d> ) {
280 return get_f3d(value);
281 }
282
283 else if constexpr ( std::is_same_v<data_t, VIPRA::f3dVec> ) {
284 return get_f3d_vec(value);
285 }
286
287 else if constexpr ( std::is_same_v<data_t, std::vector<VIPRA::Geometry::Polygon>> ) {
288 return get_polygons(value);
289 }
290
291 else if constexpr ( VIPRA::Util::is_specialization<data_t, std::vector>::value ) {
292 using value_t = typename VIPRA::Util::get_specialization_internal<data_t>::type;
293 return get_vector<value_t>(value);
294 }
295
296 else if constexpr ( VIPRA::Util::is_map_v<data_t> ) {
297 using value_t = typename VIPRA::Util::get_map_specialization<data_t>::value_t;
298 return get_map<value_t>(value);
299 }
300
301 else {
302 try {
303 return value.get().get<data_t>();
304 }
305 catch ( nlohmann::json::type_error const& e ) {
306 return std::nullopt;
307 }
308 }
309
310 return std::nullopt;
311}
